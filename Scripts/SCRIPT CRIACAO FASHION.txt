CREATE DATABASE FASHIONRECYCLE
GO

USE FASHIONRECYCLE
GO

CREATE TABLE [ROLE] (
    ID INT IDENTITY (1, 1) PRIMARY KEY,
    DESCRIPTION VARCHAR(50),
    CREATIONDATE DATETIME
)
GO

CREATE TABLE [USER] (
    ID INT IDENTITY (1, 1) PRIMARY KEY,
    USERNAME VARCHAR(20) UNIQUE,
    PASSWORD VARCHAR(100),
    NAME VARCHAR(100),
    EMAIL VARCHAR(50),
    ROLEID INT FOREIGN KEY REFERENCES [ROLE] (ID),
    ACTIVE BIT,
    CREATIONDATE DATETIME
)
GO

CREATE TABLE [CLIENT] (
	[ID] [int] IDENTITY(1,1) PRIMARY KEY,
	[NAME] [varchar](50) NULL,
    EMAIL [varchar](50) NULL,
	[PHONENUMBER] [varchar](15) NULL,
	[CPF] [varchar](11) NULL,
	[ADDRESS] [varchar](100) NULL,
	[STREETNUMBER] [varchar](10) NULL,
	[CEP] [varchar](11) NULL,
	[ACTIVE] [bit] NULL,
	[CREATIONDATE] [datetime] NULL
)
GO


CREATE TABLE [PARTNER] (
    ID INT IDENTITY (1, 1) PRIMARY KEY,
    [NAME] [varchar](50),
    EMAIL [varchar](50) NULL,
	[PHONENUMBER] [varchar](15),
	[CPF] [varchar](11),
    [CNPJ] [varchar](11),
	[ADDRESS] [varchar](100),
	[STREETNUMBER] [varchar](10),
	[CEP] [varchar](11),
	[ACTIVE] [bit],
	[CREATIONDATE] DATETIME
)
GO

CREATE TABLE [PROVIDER] (
    ID INT IDENTITY (1, 1) PRIMARY KEY,
    [COMPANYNAME] [varchar](50),
    [LEGALCOMPANEYNAME] [varchar](50),
    EMAIL [varchar](50) NULL,
	[PHONENUMBER] [varchar](15),	
    [CNPJ] [varchar](14),
	[ADDRESS] [varchar](100),
	[STREETNUMBER] [varchar](10),
	[CEP] [varchar](11),
	[ACTIVE] [bit],
	[CREATIONDATE] DATETIME
)
GO

CREATE TABLE PRODUCT (
    ID INT IDENTITY (1, 1) PRIMARY KEY,
    [NAME] [varchar](100),
    BRAND VARCHAR(100),
    AMOUNTINVENTORY INT,
    PRICEPARTNER DECIMAL,
    PRICESALE DECIMAL,
    IDPARTNER INT FOREIGN KEY REFERENCES PARTNER (ID),
	[ACTIVE] [bit],
	[CREATIONDATE] DATETIME
)
GO


CREATE TABLE PAYMENTMETHOD (
    ID INT IDENTITY (1, 1) PRIMARY KEY,
    [DESCRIPTION] VARCHAR(50),
	[ACTIVE] [bit],
	[CREATIONDATE] DATETIME
)
GO

CREATE TABLE SALES (
    ID INT IDENTITY (1, 1) PRIMARY KEY,        
    IDCLIENT INT FOREIGN KEY REFERENCES CLIENT (ID),    
    AMOUNTSALE DECIMAL,
    IDPAYMENTMETHOD INT FOREIGN KEY REFERENCES PAYMENTMETHOD (ID),
    [OBSERVATION] VARCHAR(100),
	[ACTIVE] [bit],
	[CREATIONDATE] DATETIME
)
GO

CREATE TABLE SALESITEMS (
    ID INT IDENTITY (1, 1) PRIMARY KEY,
    IDSALES INT FOREIGN KEY REFERENCES SALES (ID),        
    IDPRODUCT INT FOREIGN KEY REFERENCES PRODUCT (ID),
    IDPARTNER INT FOREIGN KEY REFERENCES [PARTNER] (ID),
    PRICESALE DECIMAL,
	[ACTIVE] [bit],
	[CREATIONDATE] DATETIME
)
GO
